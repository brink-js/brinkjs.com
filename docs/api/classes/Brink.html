    alias
    (
    • key
    )

    Defined in src/brink/utils/alias.js:14

    Alias a property to another property on the object.

    
var obj = $b.Object.create({
    a : 'test',
    b : $b.alias('a')
});

console.log(obj.a, obj.b); //test, test
this.b = 'test2';
console.log(obj.a, obj.b); // test2, test2



    
var obj = $b.Object.create({a : 'test'});
obj.prop('b', $b.alias('a'));

console.log(obj.a, obj.b); // test, test

obj.b = 'test2';

console.log(obj.a, obj.b); // test2, test2

    Function

    Defined in src/brink/utils/bindFunction.js:11

    Bind a function to a specific scope. Like Function.prototype.bind(), but does not modify the original function.

    
var obj = $b.Object.create({
    a : 'test'
});

function test () {
    console.log(this.a);
}

var boundTest = $b.bindFunction(test, obj);
boundTest(); // test

    • fn Function

      The function to bind.
